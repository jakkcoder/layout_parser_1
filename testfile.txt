import boto3
import random
import string

# Function to generate random string data
def generate_random_string(length=32):
    characters = string.ascii_letters + string.digits
    return ''.join(random.choice(characters) for _ in range(length))

# AWS S3 configuration
bucket_name = "your-bucket-name"  # Replace with your S3 bucket name
object_key = "random_data.txt"  # The key (file name) you want to use in S3

# Initialize S3 client
s3_client = boto3.client('s3')

# Generate random string data
random_data = generate_random_string(128)  # 128-character random string

# Upload the data to S3
try:
    s3_client.put_object(
        Bucket=bucket_name,
        Key=object_key,
        Body=random_data
    )
    print(f"Random string data uploaded to {bucket_name}/{object_key}")
except Exception as e:
    print(f"Error uploading data: {e}")
