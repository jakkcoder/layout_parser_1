SELECT 
  SUM(CASE WHEN APPLICATION_INSTANCE_ID LIKE '%gdt%' THEN COST ELSE 0 END) AS total_gdt_cost,
  SUM(CASE WHEN APPLICATION_INSTANCE_ID NOT LIKE '%gdt%' OR APPLICATION_INSTANCE_ID IS NULL THEN COST ELSE 0 END) AS total_non_gdt_cost,
  SUM(COST) AS total_cost
FROM 
  `hsbc-10505826-ihubhds-dev.GCE_BILLING_DATA_DEV.GDT_BILLING_view`;
