import pyautogui
import time

def prevent_sleep(interval=60):
    try:
        while True:
            pyautogui.move(0, 1)  # Move the mouse up by 1 pixel
            pyautogui.move(0, -1) # Move the mouse down by 1 pixel
            time.sleep(interval)  # Wait for the specified interval
    except KeyboardInterrupt:
        print("Program stopped by user")

if __name__ == "__main__":
    prevent_sleep()
